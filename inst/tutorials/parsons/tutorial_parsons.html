Example

Try this programming problem:

question_parsons(
  initial = c(
    "iris",
    "mutate(...)",
    "summarize(...)",
    "print()"
  ),
  answer(
    c(
      "iris",
      "mutate(...)",
      "summarize(...)"
    ),
    correct = TRUE
  ),
  answer(
    c(
      "iris",
      "mutate(...)",
      "summarize(...)",
      "print()"
    ),
    message = "You don't need print()"
  )
)
